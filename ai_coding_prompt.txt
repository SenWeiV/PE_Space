# 任务目标

根据用户提供的具体需求，生成一个完整的、可直接部署到内部工具平台的 Streamlit 应用。你必须严格遵守以下所有技术规范，确保代码能够无缝集成到平台并稳定运行。

# 平台背景

- 这是一个类似 HuggingFace Spaces 的内部工具托管平台
- 用户上传包含 `app.py` 和 `requirements.txt` 的 zip 包即可部署
- 平台自动注入 Dockerfile，**无需手动提供**
- 应用需支持多人同时使用，数据隔离且可重复运行
- 部署路径：`/app`，持久化数据目录：`/app/data`（已挂载到宿主机，跨部署保留）
- Python 版本：3.11
- 运行命令：`streamlit run app.py --server.port 8501`
- 网络：容器内可访问外部 API（OpenAI、Anthropic、自定义模型服务等）

## 三、路径与数据存储规范（重要）

### 3.1 两类目录，用途严格区分

```python
from pathlib import Path

# 当前脚本所在目录（临时文件、运行时缓存，容器重启后可能丢失）
BASE_DIR = Path(__file__).parent

# 持久化数据目录（跨部署保留，已挂载到宿主机 —— 重要数据必须写这里）
DATA_DIR = Path("/app/data")
DATA_DIR.mkdir(parents=True, exist_ok=True)  # 启动时必须确保目录存在
```

### 3.2 数据存储选择原则

| 数据类型 | 存储位置 | 说明 |
|---------|---------|------|
| 用户上传的源文件（处理前） | `DATA_DIR / "uploads"` | 持久化，跨部署保留 |
| 处理结果文件 | `DATA_DIR / "outputs"` | 持久化，供下载 |
| 历史记录、日志 | `DATA_DIR / "history"` | 持久化，可追溯 |
| 临时中间文件 | `BASE_DIR / "tmp"` | 非持久化，处理完立即删除 |
| 当次会话数据 | `st.session_state` | 仅当前用户会话，不持久化 |

### 3.3 原子化写入（多用户并发安全）

```python
import tempfile, shutil, fcntl

def atomic_write_json(filepath: Path, data: dict) -> None:
    """原子化写入 JSON，防止并发写入损坏文件"""
    filepath.parent.mkdir(parents=True, exist_ok=True)
    tmp = filepath.with_suffix(".tmp")
    try:
        with open(tmp, "w", encoding="utf-8") as f:
            fcntl.flock(f, fcntl.LOCK_EX)  # 加排他锁
            json.dump(data, f, ensure_ascii=False, indent=2)
            fcntl.flock(f, fcntl.LOCK_UN)
        shutil.move(str(tmp), str(filepath))  # 原子替换
    except Exception:
        tmp.unlink(missing_ok=True)
        raise
```

### 5.2 OpenAI 兼容接口（推荐，支持大多数国内外模型）

```python
# src/services/llm_service.py
import streamlit as st
from openai import OpenAI
import config

@st.cache_resource
def get_llm_client() -> OpenAI:
    """单例客户端，避免重复初始化"""
    return OpenAI(
        api_key=config.API_KEY,
        base_url=config.API_BASE_URL,
        timeout=config.REQUEST_TIMEOUT,
        max_retries=config.MAX_RETRIES,
    )

def call_llm(prompt: str, system: str = "", **kwargs) -> str:
    """
    调用语言模型（非流式，完整返回）

    Args:
        prompt:  用户消息内容
        system:  系统提示词（可选）
        kwargs:  覆盖默认参数（如 temperature、max_tokens）

    Returns:
        模型返回的文本内容
    """
    client = get_llm_client()
    messages = []
    if system:
        messages.append({"role": "system", "content": system})
    messages.append({"role": "user", "content": prompt})

    try:
        resp = client.chat.completions.create(
            model=config.MODEL_NAME,
            messages=messages,
            max_tokens=kwargs.get("max_tokens", config.MAX_TOKENS),
            temperature=kwargs.get("temperature", config.TEMPERATURE),
        )
        return resp.choices[0].message.content or ""
    except Exception as e:
        raise RuntimeError(f"模型调用失败：{e}") from e
```

import streamlit as st

@st.cache_resource
def get_claude_client() -> anthropic.Anthropic:

### 6.2 批量并发处理模板（线程池，适用于同步 API 调用）

```python
from concurrent.futures import ThreadPoolExecutor, as_completed
from typing import List, Dict, Any, Callable
import time

def batch_process(
    items: List[Any],
    process_fn: Callable[[Any], Any],
    max_workers: int = 5,
    progress_label: str = "处理中",
) -> tuple[List[Any], List[Dict]]:
    """
    批量并发处理，带进度条和错误收集

    Args:
        items:          待处理的数据列表
        process_fn:     处理单条数据的函数，接收一条数据，返回结果
        max_workers:    并发线程数
        progress_label: 进度条显示文字

    Returns:
        (results, errors): results 为成功结果列表，errors 为失败记录列表
    """
    results: List[Any] = []
    errors: List[Dict] = []
    total = len(items)

    progress_bar = st.progress(0, text=f"{progress_label}... 0/{total}")
    status_text = st.empty()

    with ThreadPoolExecutor(max_workers=max_workers) as executor:
        # 提交所有任务，保留原始索引
        future_to_index = {
            executor.submit(process_fn, item): (i, item)
            for i, item in enumerate(items)
        }

        completed = 0
        for future in as_completed(future_to_index):
            idx, original_item = future_to_index[future]
            completed += 1

            try:
                result = future.result()
                results.append({"index": idx, "input": original_item, "output": result})
            except Exception as e:
                errors.append({"index": idx, "input": original_item, "error": str(e)})

            # 更新进度
            pct = completed / total
            progress_bar.progress(pct, text=f"{progress_label}... {completed}/{total}，失败：{len(errors)}")
            status_text.text(f"✅ 成功：{len(results)}  ❌ 失败：{len(errors)}")

    progress_bar.empty()
    return results, errors
```

### 6.3 批量 Excel 处理完整示例

```python
def process_excel_batch(
    df: pd.DataFrame,
    input_col: str,
    output_col: str,
    concurrency: int,
    system_prompt: str = "",
) -> pd.DataFrame:
    """
    对 DataFrame 的指定列批量调用 LLM，结果写入新列

    Args:
        df:            输入 DataFrame
        input_col:     输入文本列名
        output_col:    输出结果列名
        concurrency:   并发数
        system_prompt: 系统提示词

    Returns:
        包含结果列的 DataFrame
    """
    rows = df[input_col].tolist()

    def process_one(text: str) -> str:
        if not text or str(text).strip() == "":
            return ""
        return call_llm(str(text), system=system_prompt)

    results, errors = batch_process(
        items=rows,
        process_fn=process_one,
        max_workers=concurrency,
        progress_label="批量处理",
    )

    # 按原始索引排序，填回 DataFrame
    result_map = {r["index"]: r["output"] for r in results}
    error_map  = {e["index"]: f"[错误] {e['error']}" for e in errors}

    df = df.copy()
    df[output_col] = [
        result_map.get(i, error_map.get(i, ""))
        for i in range(len(rows))
    ]

    if errors:
        st.warning(f"⚠️ {len(errors)} 行处理失败，已在结果中标注 [错误]")

    return df
```

### 6.4 批量处理注意事项

- **API 限流**：并发数建议 API 调用 5~20，大文件处理 20~100
- **错误容忍**：单行失败不影响其他行，失败记录需单独展示
- **中断恢复**：大批量任务建议分批写入 `DATA_DIR`，支持断点续传
- **超时保护**：单次调用设置合理超时（`REQUEST_TIMEOUT`），避免卡死

def read_csv_smart(file) -> pd.DataFrame:
    """自动尝试多种编码读取 CSV"""
    content = file.read()
    for encoding in ["utf-8-sig", "utf-8", "gbk", "gb2312", "latin1"]:
        try:
            return pd.read_csv(
                io.BytesIO(content),
                encoding=encoding,
                on_bad_lines="skip",
            )
